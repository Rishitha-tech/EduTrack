# -------------------------------
# Builder stage
# -------------------------------
FROM node:20-bookworm-slim AS build
WORKDIR /app

# Faster, cacheable install step
COPY package*.json ./
RUN npm ci

# Copy the rest of the source
COPY . .

# Build Next.js (shield build from missing envs like MONGODB_URI)
ENV NEXT_TELEMETRY_DISABLED=1
ARG MONGODB_URI="mongodb://127.0.0.1:27017/dummy"
ENV MONGODB_URI=$MONGODB_URI

# Create production build
RUN npm run build

# Remove devDependencies to slim down runtime
RUN npm prune --omit=dev


# -------------------------------
# Runtime stage
# -------------------------------
FROM node:20-bookworm-slim AS runner
WORKDIR /app

ENV NODE_ENV=production
ENV NEXT_TELEMETRY_DISABLED=1

# Copy only what we need to run
COPY --from=build /app/package*.json ./
COPY --from=build /app/node_modules ./node_modules
COPY --from=build /app/.next ./.next
COPY --from=build /app/public ./public
# Optional: include configs/assets your app needs at runtime
COPY --from=build /app/next.config.* ./  2>/dev/null || true
COPY --from=build /app/styles ./styles     2>/dev/null || true
COPY --from=build /app/tailwind.config.* ./ 2>/dev/null || true
COPY --from=build /app/postcss.config.* ./  2>/dev/null || true

# Render provides $PORT; expose 3000 for local runs
EXPOSE 3000
ENV PORT=3000

# Start Next.js on the provided PORT (Render sets PORT)
CMD ["sh","-c","npm run start -- -p ${PORT:-3000}"]
